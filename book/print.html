  • dans python il est possible de mettre a jour 2 variables simultann√©ment sur la meme lignes : a, b = b, a+b

Solution

def fibonacci(n):
    # retourne une liste contenant la suite de fibonacci jusqu'a n
    a = 0
    b = 1
    while b < n:
      a, b = b, a+b
      print(a)

def fibonacci(n):
    a = 0
    b = 1
    fib = [a] # Cr√©ation d'une liste fib 
    while b < n:
        a, b = b, a+b
        fib.append(a) # ajoute la nouvelle valeur de a a la fin de fib
    return fib
