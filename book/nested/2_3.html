def fibonacci(n):
    a = 0
    b = 1
    fib = [a] # Cr√©ation d'une liste fib 
    while b < n:
        a, b = b, a+b
        fib.append(a) # ajoute la nouvelle valeur de a a la fin de fib
    return fib

print(fibonacci(1000))
